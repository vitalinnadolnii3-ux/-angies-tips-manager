rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null;
    }

    function employeeProfile() {
      return get(/databases/$(database)/documents/restaurants/angies/employees/$(request.auth.uid));
    }

    function employeeExists() {
      return signedIn() &&
        exists(/databases/$(database)/documents/restaurants/angies/employees/$(request.auth.uid));
    }

    function employeeRole() {
      return employeeExists() ? employeeProfile().data.role : '';
    }

    function employeeEnabled() {
      return employeeExists() && employeeProfile().data.enabled == true;
    }

    function isAdmin() {
      return signedIn() && employeeEnabled() && employeeRole() == "Admin";
    }

    function isManagerOrAdmin() {
      return signedIn() && employeeEnabled() && (employeeRole() == "Admin" || employeeRole() == "Manager");
    }

    function isResponsible() {
      return signedIn() && employeeEnabled() && employeeRole() == "Responsible";
    }

    function canManageShifts() {
      return isManagerOrAdmin() || isResponsible();
    }

    // New helper functions reading from /users/ collection
    function userProfileExists() {
      return signedIn() &&
        exists(/databases/$(database)/documents/restaurants/angies/users/$(request.auth.uid));
    }

    function userProfile() {
      return get(/databases/$(database)/documents/restaurants/angies/users/$(request.auth.uid));
    }

    function isUserAdmin() {
      return userProfileExists() &&
        userProfile().data.role == 'admin' &&
        userProfile().data.active != false;
    }

    function isUserManager() {
      return userProfileExists() &&
        userProfile().data.role == 'manager' &&
        userProfile().data.active != false;
    }

    function isUserResponsible() {
      return userProfileExists() &&
        userProfile().data.role == 'responsible' &&
        userProfile().data.active != false;
    }

    function isUserManagerAdminOrResponsible() {
      return isUserAdmin() || isUserManager() || isUserResponsible();
    }

    function canReadAllDays() {
      return isManagerOrAdmin() || isResponsible() || isUserManagerAdminOrResponsible();
    }

    function isShiftOwner() {
      return resource.data.uid == request.auth.uid;
    }

    // Users collection - strict RBAC access control
    // Note: write access uses isAdmin() (from /employees/) during migration.
    // The first admin must be set up manually in /employees/ via Firebase Console.
    // Once /users/ docs exist, isUserAdmin() can be used for future rule updates.
    match /restaurants/angies/users/{uid} {
      // User can read their own document; admin (via employees) can read all
      allow read: if signedIn() && (request.auth.uid == uid || isAdmin());
      // Only admin can create/update user documents
      allow create, update: if isAdmin() &&
        request.resource.data.role is string &&
        request.resource.data.role in ['admin', 'manager', 'responsible', 'waiter', 'kitchen'];
      // Only admin can delete user documents
      allow delete: if isAdmin();
    }

    match /restaurants/angies/employees/{employeeId} {
      allow read: if signedIn() && employeeEnabled();
      allow create, update: if isAdmin() &&
        request.resource.data.role is string &&
        request.resource.data.role in ['Admin', 'Manager', 'Responsible', 'Waiter', 'Kitchen'];
      allow delete: if isAdmin();
    }

    match /restaurants/angies/shifts/{shiftId} {
      // All authenticated active users can view shifts.
      allow read: if signedIn() && employeeEnabled();
      allow create: if canManageShifts() &&
        request.resource.data.uid is string &&
        request.resource.data.date is string &&
        request.resource.data.weekStart is string &&
        request.resource.data.shiftType is string;
      allow update: if canManageShifts();
      allow delete: if canManageShifts();
    }

    match /restaurants/angies/days/{dayId} {
      // Admin/Manager/Responsible can read all tips and history.
      // Waiter can read only own entries (must query with uid == request.auth.uid).
      allow read: if canReadAllDays() ||
        (signedIn() &&
         employeeEnabled() &&
         resource.data.keys().hasAll(['uid']) &&
         resource.data.uid == request.auth.uid);
      allow write: if isAdmin();
    }

    match /restaurants/angies/settings/{settingId} {
      // Only admin can access settings
      allow read, write: if isAdmin();
    }

    match /restaurants/angies/chat/{chatId} {
      allow read, write: if signedIn() && employeeEnabled();
    }

    match /restaurants/angies/logs/{logId} {
      allow read: if isManagerOrAdmin();
      allow create: if signedIn() && employeeEnabled();
    }
  }
}
