rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null;
    }

    function isAdmin() {
      return signedIn() &&
        exists(/databases/$(database)/documents/restaurants/angies/employees/$(request.auth.uid)) &&
        get(/databases/$(database)/documents/restaurants/angies/employees/$(request.auth.uid)).data.role == "Admin" &&
        get(/databases/$(database)/documents/restaurants/angies/employees/$(request.auth.uid)).data.enabled == true;
    }

    match /restaurants/angies/employees/{employeeId} {
      allow read: if signedIn() && (isAdmin() || request.auth.uid == employeeId);
      allow create, update, delete: if isAdmin();
    }

    match /restaurants/angies/{document=**} {
      allow read, write: if signedIn();
    }
  }
}
